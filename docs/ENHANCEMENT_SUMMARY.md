// 多维度筛选逻辑
function applyEnhancedFilters() {
    let filtered = aiToolsData.filter(tool => {
        return matchesCategory(tool) && 
               matchesSearch(tool) && 
               matchesPricing(tool) && 
               matchesChinese(tool) && 
               matchesRating(tool) && 
               matchesPopularity(tool);
    });
    return sortTools(filtered);
}
