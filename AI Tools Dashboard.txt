    <!-- 引用生成的 JS 数据 (如未生成，下方有 Fallback 数据防止报错) -->
    <script src="js/tools_data.js"></script> 
    
    <script>
        // Fallback 数据 (如果本地没有 tools_data.js，使用此数据确保页面不白屏)
        if (typeof aiToolsData === 'undefined') {
            console.warn("未检测到 js/tools_data.js，使用临时数据。请运行 python tools_manager.py 生成真实数据。");
            var aiToolsData = [
                {
                    name: "请运行Python脚本", category: "other", desc: "运行 tools_manager.py 以获取真实数据。", 
                    url: "#", logo: "https://via.placeholder.com/100?text=Run+Script", tags: ["提示"]
                }
            ];
        }

        const grid = document.getElementById('tools-grid');
        const countEl = document.getElementById('total-count');

        function render(data) {
            grid.innerHTML = '';
            countEl.innerText = data.length;
            
            data.forEach(tool => {
                const card = document.createElement('div');
                card.className = 'tool-card bg-white rounded-xl p-4 border border-slate-100 flex flex-col cursor-pointer group relative overflow-hidden';
                
                // 处理 Logo 加载失败的情况
                const logoSrc = tool.logo || 'https://via.placeholder.com/64?text=AI';
                
                card.innerHTML = `
                    <div class="flex items-start justify-between mb-3">
                        <div class="w-12 h-12 rounded-lg bg-slate-50 border border-slate-100 flex items-center justify-center logo-container overflow-hidden">
                            <img src="${logoSrc}" alt="${tool.name}" class="w-10 h-10" onerror="this.src='https://via.placeholder.com/64?text=${tool.name[0]}'">
                        </div>
                        <div class="bg-slate-50 text-slate-500 text-xs px-2 py-1 rounded">
                            ${tool.category === 'text' ? '对话' : tool.category === 'image' ? '绘画' : tool.category}
                        </div>
                    </div>
                    <h3 class="font-bold text-slate-800 mb-1 group-hover:text-blue-600 transition-colors">${tool.name}</h3>
                    <p class="text-slate-500 text-xs leading-relaxed mb-4 line-clamp-2 h-8">${tool.desc}</p>
                    <div class="mt-auto flex items-center justify-between">
                        <div class="flex gap-1">
                            ${(tool.tags || []).slice(0, 2).map(t => `<span class="text-[10px] px-1.5 py-0.5 bg-slate-100 text-slate-500 rounded">${t}</span>`).join('')}
                        </div>
                        <button class="text-blue-600 text-xs font-medium hover:underline opacity-0 group-hover:opacity-100 transition-opacity">访问官网 &rarr;</button>
                    </div>
                `;
                
                // 真实链接跳转核心逻辑
                card.onclick = (e) => {
                    // 只有当存在有效 URL 时才跳转
                    if(tool.url && tool.url !== '#') {
                        window.open(tool.url, '_blank');
                    }
                };
                
                grid.appendChild(card);
            });
        }

        // 初始化
        render(aiToolsData);

        // 搜索逻辑
        document.getElementById('search-input').addEventListener('input', (e) => {
            const val = e.target.value.toLowerCase();
            const filtered = aiToolsData.filter(t => t.name.toLowerCase().includes(val) || t.desc.toLowerCase().includes(val));
            render(filtered);
        });

        // 简单的图表
        const ctx = document.getElementById('categoryChart').getContext('2d');
        const cats = {}; 
        aiToolsData.forEach(t => cats[t.category] = (cats[t.category]||0)+1);
        new Chart(ctx, {
            type: 'bar',
            data: {
                labels: Object.keys(cats),
                datasets: [{
                    label: '工具数量',
                    data: Object.values(cats),
                    backgroundColor: '#3b82f6',
                    borderRadius: 4
                }]
            },
            options: { responsive: true, maintainAspectRatio: false, plugins: { legend: {display:false} } }
        });
    </script>
